1. SELECT COUNT(*) AS `Загальна кількість книг`, SUM(price) AS `Загальна вартість книг`, AVG(price) AS `Середня вартість книг`, MIN(price) AS `Мінімальна ціна`, MAX(price) AS `Максимальна ціна` FROM bookshop;
2. SELECT COUNT(*) AS `Загальна кількість книг` FROM bookshop WHERE price IS NOT NULL;
3. SELECT new, COUNT(*) AS `Загальна кількість книг`, SUM(price) AS `Загальна вартість`, AVG(price) AS `Середня вартість`, MIN(price) AS `Мінімальна ціна`, MAX(price) AS `Максимальна ціна` FROM bookshop GROUP BY new;
4. SELECT YEAR(dates) AS `Рік видання`, COUNT(*) AS `Загальна кількість книг`, SUM(price) AS `Загальна вартість`, AVG(price) AS `Середня вартість`, MIN(price) AS `Мінімальна ціна`, MAX(price) AS `Максимальна ціна` FROM bookshop GROUP BY YEAR(dates);
5. SELECT YEAR(dates) AS `Рік видання`, COUNT(*) AS `Загальна кількість книг`, SUM(price) AS `Загальна вартість книг`, AVG(price) AS `Середня вартість книг`, MIN(price) AS `Мінімальна ціна`, MAX(price) AS `Максимальна ціна` FROM bookshop WHERE price < 10 OR price > 20 GROUP BY YEAR(dates);

7. SELECT COUNT(*) AS `Загальна кількість кодів книг`, COUNT(DISTINCT code) AS `Кількість унікальних кодів книг` FROM bookshop;
8. SELECT SUBSTRING(name, 1, 1) AS `Перша буква назви`, COUNT(*) AS `Загальна кількість книг`, SUM(price) AS `Загальна вартість книг` FROM bookshop GROUP BY SUBSTRING(name, 1, 1) ORDER BY `Загальна кількість книг` DESC;
9. SELECT SUBSTRING(name, 1, 1) AS `Перша буква`, COUNT(*) AS `Загальна кількість книг`, SUM(price) AS `Загальна вартість книг` FROM bookshop WHERE name REGEXP '^[^A-Za-z0-9]' GROUP BY `Перша буква`;
10. SELECT SUBSTRING(name, 1, 1) AS `Перша буква`, COUNT(*) AS `Загальна кількість книг`, SUM(price) AS `Загальна вартість книг` FROM bookshop WHERE name NOT REGEXP '^[A-Za-z0-9]' AND YEAR(dates) > 2000 GROUP BY `Перша буква` ORDER BY `Загальна кількість книг` DESC;
11. SELECT SUBSTRING(name, 1, 1) AS `Перша буква`, COUNT(*) AS `Загальна кількість`, SUM(price) AS `Загальна вартість` FROM bookshop GROUP BY `Перша буква` ORDER BY `Перша буква` DESC;

15. SELECT CASE WHEN `new` = 'Yes' THEN 'Новинка' ELSE 'Не новинка' END AS `NeW`, SUM(price) AS `Загальна ціна`, SUM(price * 36.9) AS `Загальна ціна в грн.`, SUM(price * 0.94) AS `Загальна ціна в євро` FROM bookshop WHERE dates IS NOT NULL GROUP BY `NeW`;
